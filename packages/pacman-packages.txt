# Unlike Ubuntu/apt, the Arch official repos carry recent versions of
# starship, zoxide, uv, ruff, lazygit and go, so they come from pacman here
# instead of the curl-based installers used on Debian.

zsh
tmux
git
curl
wget
unzip
base-devel

# CLI tools
fzf
fd
bat
btop
direnv
# nyancat — installed best-effort by set-pacman-packages (its own transaction),
# so a slow mirror on this toy can't abort the essential packages.
starship
zoxide
uv
ruff
lazygit
go

nodejs
npm
